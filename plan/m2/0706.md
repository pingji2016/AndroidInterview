```kotlin
// app/src/main/java/com/example/androidagent/MyApplication.kt

package com.example.androidagent

import android.app.Application
import com.example.androidagent.agent.ReActAgent
import com.example.androidagent.inference.LlamaInference
import com.example.androidagent.tool.ToolManager
import com.example.androidagent.tool.impl.*

class MyApplication : Application() {

    lateinit var agent: ReActAgent
        private set
    lateinit var toolManager: ToolManager
        private set
    lateinit var inference: LlamaInference
        private set

    override fun onCreate() {
        super.onCreate()
        
        // 初始化推理引擎
        inference = LlamaInference(this)
        
        // 初始化 Tool 系统
        toolManager = ToolManager().apply {
            register(SendNotificationTool(this@MyApplication))
            register(AdjustBrightnessTool(this@MyApplication))
            register(GetCurrentAppTool(this@MyApplication))
            register(SetVolumeTool(this@MyApplication))
            register(OpenAppTool(this@MyApplication))
        }
        
        // 初始化 Agent
        agent = ReActAgent(
            inference = inference,
            toolManager = toolManager
        )

        // 创建通知渠道
        SendNotificationTool.createNotificationChannel(this)
    }
}
```

### ChatViewModel 集成 Agent

```kotlin
// app/src/main/java/com/example/androidagent/ui/ChatViewModel.kt

package com.example.androidagent.ui

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.example.androidagent.agent.ReActAgent
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch

class ChatViewModel(
    private val agent: ReActAgent
) : ViewModel() {

    private val _messages = MutableStateFlow<List<ChatMessage>>(emptyList())
    val messages: StateFlow<List<ChatMessage>> = _messages.asStateFlow()

    private val _isLoading = MutableStateFlow(false)
    val isLoading: StateFlow<Boolean> = _isLoading.asStateFlow()

    init {
        _messages.value = listOf(
            ChatMessage(
                content = "Agent 已就绪！\n\n可用的指令示例：\n" +
                    "• \"发通知说：写完了\"\n" +
                    "• \"调亮屏幕到 80%\"\n" +
                    "• \"打开微信并调高音量\"\n" +
                    "• \"当前在哪个 App？\"",
                isUser = false
            )
        )
        
        // 后台加载模型
        viewModelScope.launch {
            agent.loadModel()
        }
    }

    fun sendMessage(text: String) {
        _messages.value = _messages.value + ChatMessage(
            content = text,
            isUser = true
        )

        viewModelScope.launch {
            _isLoading.value = true

            val startTime = System.currentTimeMillis()

            // 执行 ReAct 循环
            val result = agent.execute(
                userInput = text,
                onStep = { step ->
                    // 可选：在 UI 上显示思考过程
                    _messages.value = _messages.value + ChatMessage(
                        content = "→ ${step.thought}\n  执行: ${step.action}\n  结果: ${step.observation}",
                        isUser = false
                    )
                }
            )

            val elapsed = System.currentTimeMillis() - startTime

            _messages.value = _messages.value + ChatMessage(
                content = result,
                isUser = false,
                inferenceTimeMs = elapsed
            )

            _isLoading.value = false
        }
    }
}
```

### 完整的错误恢复策略

```kotlin
/**
 * ReActAgent.execute() 的完整错误处理版本
 */
suspend fun execute(userInput: String): String {
    val history = mutableListOf<ReActStep>()
    var consecutiveParseFailures = 0  // 连续解析失败计数

    for (step in 1..MAX_STEPS) {
        // 1. 构建 Prompt
        val prompt = buildReActPrompt(userInput, toolManager.getAllSchemas(), history)

        // 2. LLM 推理
        val output = inference.runInference(prompt).getOrElse { error ->
            // 推理失败，重试
            history.add(ReActStep(
                thought = "推理异常",
                action = "retry",
                actionInput = error.message ?: "未知错误",
                observation = "推理异常，重试中"
            ))
            continue
        }

        // 3. 解析输出
        val parsed = parser.parse(output)
        
        if (parsed == null) {
            consecutiveParseFailures++
            if (consecutiveParseFailures >= 3) {
                // 连续 3 次解析失败，降级为纯文本回复
                return "抱歉，无法理解你的指令，请换一种方式描述。\n\n" +
                    "可尝试：\n" +
                    "• \"发通知说：完成\"\n" +
                    "• \"调亮屏幕\"\n" +
                    "• \"打开微信\""
            }
            history.add(ReActStep(
                thought = "解析失败",
                action = "retry",
                actionInput = output,
                observation = "请严格按照 JSON 格式回复"
            ))
            continue
        }
        
        consecutiveParseFailures = 0  // 重置

        // 4. Final Answer
        if (parsed.action == "Final Answer") {
            return parsed.params["answer"] as? String
                ?: "任务完成"
        }

        // 5. 执行工具
        val result = toolManager.execute(parsed)

        // 6. 记录步骤
        history.add(ReActStep(
            thought = parsed.params["thought"] as? String ?: "",
            action = parsed.action,
            actionInput = parsed.params.toString(),
            observation = result.toObservation()
        ))
    }

    return "执行步骤过多，请简化指令重试"
}
```

## 四、上下文管理

### 问题：每次推理都是独立的

当前实现的问题是：每次调用 `agent.execute()` 都是全新的 ReAct 循环，**不记得上一轮对话**。

```
用户: "打开微信"          → Agent: 打开了 ✅
用户: "给我放首歌"        → Agent: ...不知道在哪个 App 里放
```

### 解决方案：维护对话历史

```kotlin
// app/src/main/java/com/example/androidagent/agent/ConversationManager.kt

package com.example.androidagent.agent

/**
 * 对话历史管理器
 *
 * 维护多轮对话的上下文，让 Agent 能记住之前的对话
 */
class ConversationManager(
    private val maxHistoryTurns: Int = 3  // 保留最近 3 轮
) {

    private val history = mutableListOf<ConversationTurn>()

    /**
     * 添加一轮对话
     */
    fun addTurn(userInput: String, agentResponse: String) {
        history.add(ConversationTurn(userInput, agentResponse))
        // 超出上限时删除最旧的
        if (history.size > maxHistoryTurns) {
            history.removeAt(0)
        }
    }

    /**
     * 构建上下文字符串（拼入 Prompt）
     */
    fun buildContext(): String {
        if (history.isEmpty()) return ""
        
        return history.joinToString("\n") { turn ->
            """
用户: ${turn.userInput}
助手: ${turn.agentResponse}
            """.trimIndent()
        }
    }

    /**
     * 清空历史
     */
    fun clear() {
        history.clear()
    }

    data class ConversationTurn(
        val userInput: String,
        val agentResponse: String
    )
}
```

```kotlin
        )

        conversationManager.clear()

}
```

### 更新 Prompt 模板（带上下文）

```kotlin
fun buildReActPrompt(
    userInput: String,
    toolSchemas: List<ToolSchema>,
    history: List<ReActStep> = emptyList(),
    conversationContext: String = ""
): String {
    val toolDesc = toolSchemas.joinToString("\n") { schema ->
        val params = schema.parameters.joinToString(", ") { p ->
            "${p.name} (${p.type}): ${p.description}"
        }
        "- ${schema.name}: ${schema.description}，参数: [$params]"
    }

    val historyStr = history.joinToString("\n") { step ->
        """
Thought: ${step.thought}
Action: ${step.action}
Action Input: ${step.actionInput}
Observation: ${step.observation}
        """.trimIndent()
    }

    return """
<|im_start|>system
你是一个 Android 手机上的 AI 助手。你需要通过思考(Thought)和行动(Action)来完成用户的指令。

可用的工具：
$toolDesc

回复格式要求（JSON）：
{
  "thought": "你的思考过程",
  "action": "工具名称 或 Final Answer",
  "params": { ... },
  "answer": "如果 action 是 Final Answer，这里填写回复内容"
}
<|im_end|>

${if (conversationContext.isNotBlank()) """
<|im_start|>conversation
之前的对话：
$conversationContext
<|im_end|>
""" else ""}

${if (historyStr.isNotBlank()) """
<|im_start|>history
当前步骤历史：
$historyStr
<|im_end|>
""" else ""}

<|im_start|>user
$userInput
<|im_end|>
<|im_start|>assistant
    """.trimIndent()
}
```
