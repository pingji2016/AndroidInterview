### CMakeLists.txt 配置

```cmake
# app/src/main/jni/CMakeLists.txt
cmake_minimum_required(VERSION 3.18)

project("llama")

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 添加核心推理引擎源码
add_library(llama STATIC
    ${CMAKE_SOURCE_DIR}/../cpp/llama.cpp
    ${CMAKE_SOURCE_DIR}/../cpp/ggml.c
    ${CMAKE_SOURCE_DIR}/../cpp/ggml-alloc.c
    ${CMAKE_SOURCE_DIR}/../cpp/ggml-backend.c
)

# 添加 JNI 桥接库
add_library(llama-android SHARED
    llama-android.cpp
    simple_chat.cpp
)

# 链接
target_link_libraries(llama-android
    llama
    jnigraphics
    log
)

# 包含头文件
target_include_directories(llama-android PRIVATE
    ${CMAKE_SOURCE_DIR}/../cpp
)
```

## 三、封装 LlamaInference 类

### Kotlin 封装（面向协程）

```kotlin
// app/src/main/java/com/example/androidagent/inference/LlamaInference.kt

package com.example.androidagent.inference

import android.content.Context
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import java.io.File

/**
 * llama.cpp 推理封装类
 *
 * 职责：
 * 1. 管理模型的生命周期（加载/卸载）
 * 2. 提供 suspend 挂起函数，不阻塞主线程
 * 3. 解析模型输出
 */
class LlamaInference(private val context: Context) {

    companion object {
        private const val TAG = "LlamaInference"
        private const val MODEL_FILENAME = "tinyllama-1.1b-chat-v1.0.Q4_K_M.gguf"
        // 最大 token 数：控制回复长度
        private const val MAX_TOKENS = 256
        // 线程数：建议设为手机 CPU 核心数
        private val THREAD_COUNT = Runtime.getRuntime().availableProcessors()
    }

    // 模型是否已加载
    private var isLoaded = false
    
    // 模型文件路径（assets 中的模型需要先复制到内部存储）
    private var modelPath: String? = null
        private set

    /**
     * 加载模型（耗时操作，需要在 IO 线程执行）
     *
     * 流程：
     * 1. 将模型从 assets 复制到内部存储（首次需要）
     * 2. 调用 native 方法加载模型
     */
    suspend fun loadModel(): Result<Unit> = withContext(Dispatchers.IO) {
        try {
            // 1. 准备模型文件
            val modelFile = prepareModelFile()
            
            // 2. 调用 JNI 加载模型
            // (nativeLoadModel 会在后续步骤实现)
            val result = nativeLoadModel(modelFile.absolutePath)
            
            if (result == 0) {
                isLoaded = true
                modelPath = modelFile.absolutePath
                Result.success(Unit)
            } else {
                Result.failure(RuntimeException("Failed to load model, error code: $result"))
            }
        } catch (e: Exception) {
            Result.failure(e)
        }
    }

    /**
     * 执行推理（耗时操作）
     *
     * @param input 用户输入的文本
     * @return 模型生成的回复
     */
    suspend fun runInference(input: String): Result<String> = withContext(Dispatchers.IO) {
        if (!isLoaded) {
            return@withContext Result.failure(IllegalStateException("Model not loaded"))
        }
        
        try {
            // 构建 Prompt（Chat 格式）
            val prompt = buildChatPrompt(input)
            
            // 调用 JNI 推理
            val output = nativeRunInference(prompt, MAX_TOKENS, THREAD_COUNT)
            
            // 清洗输出（去掉特殊 token）
            val cleaned = cleanOutput(output)
            
            Result.success(cleaned)
        } catch (e: Exception) {
            Result.failure(e)
        }
    }

    /**
     * 卸载模型，释放内存
     */
    suspend fun unloadModel(): Result<Unit> = withContext(Dispatchers.IO) {
        try {
            nativeUnloadModel()
            isLoaded = false
            modelPath = null
            Result.success(Unit)
        } catch (e: Exception) {
            Result.failure(e)
        }
    }

    // ========== 私有方法 ==========

    private fun prepareModelFile(): File {
        val internalDir = File(context.filesDir, "models")
        if (!internalDir.exists()) internalDir.mkdirs()
        
        val modelFile = File(internalDir, MODEL_FILENAME)
        
        // 首次运行时将 assets 中的模型复制到内部存储
        if (!modelFile.exists()) {
            context.assets.open(MODEL_FILENAME).use { input ->
                modelFile.outputStream().use { output ->
                    input.copyTo(output)
                }
            }
        }
        
        return modelFile
    }

    /**
     * 构建 Chat Prompt
     * TinyLlama Chat 格式：
     * <|im_start|>user
     * {用户输入}<|im_end|>
     * <|im_start|>assistant
     */
    private fun buildChatPrompt(userInput: String): String {
        return """
            <|im_start|>system
            You are a helpful Android assistant. You can control system functions.
            <|im_end|>
            <|im_start|>user
            $userInput<|im_end|>
            <|im_start|>assistant
        """.trimIndent()
    }

    /**
     * 清洗模型输出
     * 去掉 <|im_end|>、<|im_start|> 等特殊 token
     */
    private fun cleanOutput(output: String): String {
        return output
            .replace("<|im_end|>", "")
            .replace("<|im_start|>", "")
            .replace("assistant", "")
            .trim()
    }

    // ========== JNI Native 方法 ==========

    /**
     * 加载模型
     * @param modelPath 模型文件绝对路径
     * @return 0 成功，非 0 失败
     */
    private external fun nativeLoadModel(modelPath: String): Int

    /**
     * 执行推理
     * @param prompt 输入 prompt
     * @param maxTokens 最大生成 token 数
     * @param threads 线程数
     * @return 生成的文本
     */
    private external fun nativeRunInference(prompt: String, maxTokens: Int, threads: Int): String

    /**
     * 卸载模型
     */
    private external fun nativeUnloadModel()
}
```

### 使用示例（在 ViewModel 中调用）

```kotlin
// 在 ViewModel 中使用
class ChatViewModel(private val inference: LlamaInference) : ViewModel() {

    private val _response = MutableStateFlow<String?>(null)
    val response: StateFlow<String?> = _response.asStateFlow()

    private val _isLoading = MutableStateFlow(false)
    val isLoading: StateFlow<Boolean> = _isLoading.asStateFlow()

    fun sendMessage(message: String) {
        viewModelScope.launch {
            _isLoading.value = true
            _response.value = null

            inference.runInference(message)
                .onSuccess { result ->
                    _response.value = result
                }
                .onFailure { error ->
                    _response.value = "Error: ${error.message}"
                }

            _isLoading.value = false
        }
    }
}
```
